```notes
class FilePicker extends React.Component {
  constructor (props){
    super(props);
    this.state = {
      file: null
    }
  }
  pickFile (e) {
    this.setState({file: e.target.files[0]})
  }
  render() {
    return (
      <div>
        <input
          type="file"
          onChange={(e) => this.pickFile(e)}
        />
        {
          this.state.file ?
          <div>
            <img
              width="300"
              height="300"
              src={URL.createObjectURL(this.state.file)} />
          </div> :
          null
        }
      </div>
    )
  }
}


```
